//open the console
truffle console --network hdwallet


// Get contract
let instance = await IPFSStorage.deployed()

// Get accounts from HD Wallet
let accounts = await web3.eth.getAccounts()
console.log("Using account:", accounts[0])

// Store hash
await instance.setHash("QmbFMke1KXqnYyBBWxB74N4c5SBnJMVAiMNRcGu6x1AwQH", {from: accounts[2]})

// Retrieve hash
let hash = await instance.getHash()
console.log("Stored hash:", hash)


truffle migrate --network hdwallet --reset
